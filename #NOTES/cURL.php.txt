// Inicia o cURL
$ch = curl_init();

// Configura as opções do cURL
$ch_url = 'http://localhost/QUERY/api.php';
$ch_url .= '?';
$ch_url .= http_build_query(array(
   'class' => 'init',
   'function' => 'status',
   'data' => array(
      true,
      12,
   ),
), 'VARIABLE_', '&');

$ch_headers = array(
   'Content-type: charset=UTF-8', // application/json - text/plain; 
   'Content-length: 100',
);

$ch_username = 'root';

$ch_password = 12345678;

// Define a URL
curl_setopt($ch, CURLOPT_URL, $ch_url);
// Retorna a resposta como uma string
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true); 
// Ignora a verificação do certificado SSL
curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
// Versão do HTTP
curl_setopt($ch, CURLOPT_HTTP_VERSION,  CURL_HTTP_VERSION_1_0);
// Define porta alternativa
curl_setopt($ch, CURLOPT_PORT,  8080);
// Define o protocolo 
curl_setopt($ch, CURLOPT_PROTOCOLS,  CURLPROTO_HTTP);
// Define em milisegundos tempo maximo para a execução 
curl_setopt($ch, CURLOPT_TIMEOUT_MS,  6000);
// Tipo de autenticação
curl_setopt($ch, CURLOPT_HTTPAUTH, CURLAUTH_BASIC);
// Username
curl_setopt($ch, CURLOPT_USERNAME,  $ch_username);
// Password
curl_setopt($ch, CURLOPT_PASSWORD,  $ch_password);
// Define Headers
curl_setopt($ch, CURLOPT_HTTPHEADER,  $ch_headers);
// Define a requisição como GET
curl_setopt($ch, CURLOPT_HTTPGET,  false);
// Define a requisição como POST
curl_setopt($ch, CURLOPT_POST,  true);

// Resposta da solicitação
$ch_response = curl_exec($ch);

// Fecha a sessão cURL
curl_close($ch);
